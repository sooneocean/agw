**Mocking pattern (shared across all 3 adapter tests):**

```typescript
import { vi, describe, it, expect, beforeEach } from 'vitest';
import { EventEmitter } from 'node:events';
import type { ChildProcess } from 'node:child_process';

// Mock spawn before importing adapter
const mockSpawn = vi.fn();
vi.mock('node:child_process', () => ({
  spawn: (...args: unknown[]) => mockSpawn(...args),
  exec: vi.fn(),
}));

function createMockProcess(exitCode: number, stdout: string, stderr = ''): ChildProcess {
  const proc = new EventEmitter() as ChildProcess;
  proc.stdout = new EventEmitter() as any;
  proc.stderr = new EventEmitter() as any;
  proc.stdin = { write: vi.fn(), end: vi.fn() } as any;
  proc.pid = 12345;
  proc.killed = false;
  proc.kill = vi.fn();

  // Schedule data + close events
  queueMicrotask(() => {
    if (stdout) proc.stdout!.emit('data', Buffer.from(stdout));
    if (stderr) proc.stderr!.emit('data', Buffer.from(stderr));
    queueMicrotask(() => proc.emit('close', exitCode));
  });

  return proc;
}
```

```typescript
private updateScore(agentId: string, category: string, success: boolean, durationMs: number, cost: number): void {
  const key = `${agentId}:${category}`;
  const existing = this.scores.get(key) ?? {
    agentId, category, successCount: 0, failCount: 0,
    avgDurationMs: 0, totalCost: 0, score: 0,
  };
  if (success) existing.successCount++; else existing.failCount++;
  const total = existing.successCount + existing.failCount;
  existing.avgDurationMs = ((existing.avgDurationMs * (total - 1)) + durationMs) / total;
  existing.totalCost += cost;
  existing.score = this.calculateScore(existing);
  this.scores.set(key, existing);
  this.persist(key, existing);
}
```

- [ ] **Step 6: Update getBestAgent() to support string[]**

```typescript
getBestAgent(categoryOrCategories: string | string[]): string | undefined {
  const categories = Array.isArray(categoryOrCategories)
    ? categoryOrCategories
    : [categoryOrCategories];

  // Aggregate scores across matching categories
  const agentTotals = new Map<string, { score: number; weight: number }>();
  for (const category of categories) {
    for (const [key, entry] of this.scores.entries()) {
      if (key.endsWith(`:${category}`) && (entry.successCount + entry.failCount) >= 3) {
        const agentId = key.split(':')[0];
        const existing = agentTotals.get(agentId) ?? { score: 0, weight: 0 };
        const taskCount = entry.successCount + entry.failCount;
        existing.score += entry.score * taskCount;
        existing.weight += taskCount;
        agentTotals.set(agentId, existing);
      }
    }
  }

  let bestAgent: string | undefined;
  let bestScore = -1;
  for (const [agentId, { score, weight }] of agentTotals) {
    const weightedScore = weight > 0 ? score / weight : 0;
    if (weightedScore > bestScore) {
      bestScore = weightedScore;
      bestAgent = agentId;
    }
  }
  return bestAgent;
}
```

```typescript
private async executeDebate(comboId: string, request: CreateComboRequest): Promise<void> {
  const steps = request.steps;
  if (steps.length < 2) throw new Error('Debate requires at least 2 steps');

  const debaterSteps = steps.slice(0, -1);
  const judgeStep = steps[steps.length - 1];
  const stepResults: Record<number, string> = {};

  // Phase 1: parallel debaters — each only sees {{input}}
  // Note: status already set to 'running' by runCombo() caller — do NOT set again
  const debaterPromises = debaterSteps.map(async (step, i) => {
    if (this.isCancelled(comboId)) throw new Error('Combo cancelled');
    this.auditRepo.log(null, 'combo.step', { comboId, step: i, agent: step.agent, role: step.role, phase: 'debate' });
    const prompt = interpolate(step.prompt, { input: request.input, stepResults: {}, prev: '' });
    const task = await this.executeStep(comboId, step.agent, prompt, request);
    this.comboRepo.addTaskId(comboId, task.taskId);
    return { step: i, task };
  });

  const settled = await Promise.allSettled(debaterPromises);

  let successCount = 0;
  for (const [i, outcome] of settled.entries()) {
    if (outcome.status === 'fulfilled' && outcome.value.task.status === 'completed') {
      stepResults[i] = outcome.value.task.result?.stdout ?? '';
      this.comboRepo.setStepResult(comboId, i, stepResults[i]);
      successCount++;
    } else {
      stepResults[i] = `[ERROR: Debater ${i} failed]`;
      this.comboRepo.setStepResult(comboId, i, stepResults[i]);
      this.auditRepo.log(null, 'combo.partial', { comboId, step: i, error: 'debater failed' });
    }
  }

  if (successCount === 0) {
    throw new Error('All debaters failed');
  }

  // Phase 2: judge — sees all debater results
  if (this.isCancelled(comboId)) throw new Error('Combo cancelled');
  this.auditRepo.log(null, 'combo.step', { comboId, step: debaterSteps.length, agent: judgeStep.agent, role: judgeStep.role, phase: 'judge' });
  const judgePrompt = interpolate(judgeStep.prompt, { input: request.input, stepResults, prev: '' });
  const judgeTask = await this.executeStep(comboId, judgeStep.agent, judgePrompt, request);
  this.comboRepo.addTaskId(comboId, judgeTask.taskId);

  const finalOutput = judgeTask.result?.stdout ?? '';
  this.comboRepo.setStepResult(comboId, debaterSteps.length, finalOutput);
  this.comboRepo.setFinalOutput(comboId, finalOutput);
}
```

```typescript
function chunkArray<T>(arr: T[], size: number): T[][] {
  const chunks: T[][] = [];
  for (let i = 0; i < arr.length; i += size) {
    chunks.push(arr.slice(i, i + size));
```
